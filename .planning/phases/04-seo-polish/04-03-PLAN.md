Create a reusable SkeletonLoader component for loading states.

Component should:
- Accept variant prop: "text" | "card" | "image" | "section"
- Accept optional className prop for customization
- Render animated skeleton with pulse effect
- Use Tailwind animate-pulse utility

Structure:
```tsx
interface SkeletonLoaderProps {
  variant?: "text" | "card" | "image" | "section";
  className?: string;
}

export function SkeletonLoader({ variant = "text", className = "" }: SkeletonLoaderProps) {
  const baseClasses = "animate-pulse bg-gray-200 rounded";
  
  const variantClasses = {
    text: "h-4 w-full",
    card: "h-64 w-full",
    image: "h-96 w-full",
    section: "h-screen w-full",
  };

  return (
    <div className={`${baseClasses} ${variantClasses[variant]} ${className}`} />
  );
}
```

Keep it simple - just a div with animated background. Can be composed to create complex skeleton layouts.

```tsx
'use client'; // Add if not already present

import { useState, useEffect } from 'react';
import { SkeletonLoader } from '@/components/ui/SkeletonLoader';

export default function Home() {
  const [mounted, setMounted] = useState(false);

  useEffect(() => {
    setMounted(true);
  }, []);

  return (
    <>
      {/* Hero section - always visible */}
      <HeroMasked />
      
      {/* Below-fold sections with skeleton */}
      {!mounted ? (
        <SkeletonLoader variant="section" />
      ) : (
        <ValueCarousel />
      )}
      
      {/* Repeat for 1-2 more sections */}
    </>
  );
}
```

Note: This is a simple implementation. For production, consider using Suspense boundaries or more sophisticated loading strategies. This satisfies POLISH-02 requirement for skeleton screens.
