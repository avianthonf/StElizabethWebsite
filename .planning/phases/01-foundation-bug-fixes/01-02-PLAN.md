import { test, expect } from '@playwright/test';

test.describe('Mobile Menu', () => {
  test.beforeEach(async ({ page }) => {
    await page.goto('/');
    // Set viewport to mobile size
    await page.setViewportSize({ width: 375, height: 667 });
  });

  test('opens when hamburger button clicked', async ({ page }) => {
    // Find and click hamburger button
    const menuButton = page.getByLabel('Toggle menu');
    await expect(menuButton).toBeVisible();
    
    await menuButton.click();
    
    // Verify menu overlay is visible
    const mobileMenu = page.locator('[role="dialog"]').or(page.locator('nav').filter({ hasText: 'About Us' }));
    await expect(mobileMenu).toBeVisible();
    
    // Verify hamburger icon changed to X
    await expect(menuButton).toHaveAttribute('aria-expanded', 'true');
  });

  test('closes when X button clicked', async ({ page }) => {
    // Open menu
    const menuButton = page.getByLabel('Toggle menu');
    await menuButton.click();
    
    // Wait for menu to be visible
    const mobileMenu = page.locator('nav').filter({ hasText: 'About Us' });
    await expect(mobileMenu).toBeVisible();
    
    // Click X button to close
    await menuButton.click();
    
    // Verify menu is hidden
    await expect(mobileMenu).not.toBeVisible();
    await expect(menuButton).toHaveAttribute('aria-expanded', 'false');
  });

  test('closes when Escape key pressed', async ({ page }) => {
    // Open menu
    const menuButton = page.getByLabel('Toggle menu');
    await menuButton.click();
    
    // Wait for menu to be visible
    const mobileMenu = page.locator('nav').filter({ hasText: 'About Us' });
    await expect(mobileMenu).toBeVisible();
    
    // Press Escape key
    await page.keyboard.press('Escape');
    
    // Verify menu is hidden
    await expect(mobileMenu).not.toBeVisible();
  });

  test('prevents background scroll when open (iOS Safari)', async ({ page, browserName }) => {
    // This test is most important for WebKit (Safari)
    test.skip(browserName !== 'webkit', 'iOS Safari scroll lock test');
    
    // Scroll down the page first
    await page.evaluate(() => window.scrollTo(0, 500));
    const initialScrollY = await page.evaluate(() => window.scrollY);
    expect(initialScrollY).toBeGreaterThan(0);
    
    // Open mobile menu
    const menuButton = page.getByLabel('Toggle menu');
    await menuButton.click();
    
    // Try to scroll the page (should be locked)
    await page.mouse.wheel(0, 500);
    await page.waitForTimeout(100); // Wait for scroll attempt
    
    // Verify scroll position hasn't changed
    const scrollYAfterWheel = await page.evaluate(() => window.scrollY);
    expect(scrollYAfterWheel).toBe(initialScrollY);
    
    // Close menu
    await menuButton.click();
    
    // Verify scrolling works again
    await page.mouse.wheel(0, 500);
    await page.waitForTimeout(100);
    const scrollYAfterClose = await page.evaluate(() => window.scrollY);
    expect(scrollYAfterClose).toBeGreaterThan(initialScrollY);
  });

  test('is keyboard navigable', async ({ page }) => {
    // Tab to menu button
    await page.keyboard.press('Tab');
    await page.keyboard.press('Tab'); // May need multiple tabs depending on page structure
    
    // Open menu with Enter
    const menuButton = page.getByLabel('Toggle menu');
    await menuButton.focus();
    await page.keyboard.press('Enter');
    
    // Verify menu opened
    const mobileMenu = page.locator('nav').filter({ hasText: 'About Us' });
    await expect(mobileMenu).toBeVisible();
    
    // Tab through menu items
    await page.keyboard.press('Tab');
    const firstLink = page.locator('nav a').first();
    await expect(firstLink).toBeFocused();
    
    // Navigate with Enter key
    await page.keyboard.press('Enter');
    
    // Verify navigation occurred (URL changed or submenu opened)
    // This depends on menu structure - adjust as needed
  });

  test('has proper ARIA attributes', async ({ page }) => {
    const menuButton = page.getByLabel('Toggle menu');
    
    // Check initial state
    await expect(menuButton).toHaveAttribute('aria-label', 'Toggle menu');
    await expect(menuButton).toHaveAttribute('aria-expanded', 'false');
    
    // Open menu
    await menuButton.click();
    
    // Check open state
    await expect(menuButton).toHaveAttribute('aria-expanded', 'true');
    
    // Verify menu has proper role or aria attributes
    const mobileMenu = page.locator('nav').filter({ hasText: 'About Us' });
    await expect(mobileMenu).toBeVisible();
  });

  test('navigation links work in mobile menu', async ({ page }) => {
    // Open menu
    const menuButton = page.getByLabel('Toggle menu');
    await menuButton.click();
    
    // Wait for menu to be visible

    await expect(mobileMenu).toBeVisible();
